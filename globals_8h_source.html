165 inline int myrand( int min, int max )
166 {
167  if( min == max ) return min;
168  if( min > max )
169  {
170  int temp = min;
171  min = max;
172  max = temp;
173  }
174 
175  double gap = max - min + 1;
176  double rrr = (double)RAND_MAX / gap;
177 
178  double rand_result;
179 
180  rand_result = (double)rand() / rrr;
181 
182  if( (int)( min + (int)rand_result ) < min ) return min;
183  if( (int)( min + (int)rand_result ) > max ) return max;
184 
185  return (int)( min + (int)rand_result );
186 };
